### 10. Najdete kdy a kde mela voda zákal 8 a kdy víc než 100.
### 	Pro hledaní přesně hodnoty podle hodnot mimo PRIMARY KEY, mužů použít INDEX, ale porovnávaní musím použit ALLOW FILTERING.

cassandra@cqlsh:water_weather_sensors>

	CREATE INDEX ON water (turbidity);
	SELECT beach_name, year, month, day, time, turbidity FROM water WHERE turbidity = 8;
	SELECT beach_name, year, month, day, time, turbidity FROM water WHERE turbidity > 100 LIMIT 10 ALLOW FILTERING;


### 11. Vypište údaje o dešti ze dne 26.03.2020 na stanici Oak Street.

### 13. Vytvořte funkci: Obecná zdánlivá teplota
### 	Ukázka pro vytvorení vlastních funkci

cassandra@cqlsh:water_weather_sensors>

	CREATE OR REPLACE FUNCTION ga_temp (t DOUBLE, h INT, w DOUBLE) RETURNS NULL ON NULL INPUT RETURNS DOUBLE LANGUAGE java AS 'return t + 0.33 * (h / 100) * 6.105 * Math.exp((17.27 * t) / (t + 237.7)) - 0.7 * w - 4;';
	SELECT station_name, measurement_timestamp, air_temperature, ga_temp (air_temperature, humidity, wind_speed) AS general_apparent_temperature FROM weather WHERE station_name = 'Foster' AND measurement_timestamp > '2015-09-19 08:00:00' AND measurement_timestamp < '2015-09-19 16:00:00';
